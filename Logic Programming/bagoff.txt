% Questão 1 da Lista 6: Maiores([1,5,7],5,N) -> N = [6,7]
maior([],[],_).
maior([X|Y],[X|Z],N) :- X > N, maior(Y,Z,N),!.
maior([_|Y],Z,N) :- maior(Y,Z,N).
quest_1(Num,Lista,Z) :-
    nl,write('Elementos maiores do que '),
    write(Num),write(' na lista '),write(Lista),
    write(' : '),maior(Lista, Z, Num),nl.
